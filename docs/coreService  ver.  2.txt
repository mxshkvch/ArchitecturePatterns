components:
  schemas:
    AccountResponse:
      type: object
      description: Информация о банковском счете
      properties:
        id:
          type: string
          format: uuid
          description: Уникальный идентификатор счета
          example: "123e4567-e89b-12d3-a456-426614174000"
        accountNumber:
          type: string
          description: Номер счета (20 цифр)
          example: "12345678901234567890"
          pattern: '^\d{20}$'
        userId:
          type: string
          format: uuid
          description: ID владельца счета
          example: "123e4567-e89b-12d3-a456-426614174001"
        balance:
          type: number
          format: decimal
          description: Текущий баланс
          example: 1500.50
        currency:
          type: string
          description: Валюта счета
          example: "USD"
          enum: [USD, EUR, RUB, GBP]
        status:
          type: string
          description: Статус счета
          enum: [ACTIVE, BLOCKED, CLOSED, FROZEN]
          example: "ACTIVE"
        createdAt:
          type: string
          format: date-time
          description: Дата создания
          example: "2024-01-01T10:00:00Z"
        closedAt:
          type: string
          format: date-time
          nullable: true
          description: Дата закрытия (если счет закрыт)
          example: null

    CreateAccountRequest:
      type: object
      description: Запрос на создание нового счета
      properties:
        initialDeposit:
          type: number
          format: decimal
          description: Начальный депозит
          example: 100.00
          minimum: 0
        currency:
          type: string
          description: Валюта счета
          example: "USD"
          enum: [USD, EUR, RUB, GBP]
      required:
        - currency

    TransferRequest:
      type: object
      description: Запрос на перевод средств
      properties:
        amountMoney:
          type: number
          format: decimal
          description: Сумма перевода
          example: 500.00
          minimum: 0.01
      required:
        - amountMoney

    DepositRequest:
      type: object
      description: Запрос на пополнение счета
      properties:
        amount:
          type: number
          format: decimal
          description: Сумма депозита
          example: 1000.00
          minimum: 0.01
        description:
          type: string
          description: Описание транзакции
          example: "ATM deposit"
          maxLength: 500
      required:
        - amount

    WithdrawalRequest:
      type: object
      description: Запрос на снятие средств
      properties:
        amount:
          type: number
          format: decimal
          description: Сумма снятия
          example: 200.00
          minimum: 0.01
        description:
          type: string
          description: Описание транзакции
          example: "ATM withdrawal"
          maxLength: 500
      required:
        - amount

    TransactionResponse:
      type: object
      description: Информация о транзакции
      properties:
        id:
          type: string
          format: uuid
          description: Уникальный идентификатор транзакции
          example: "123e4567-e89b-12d3-a456-426614174002"
        accountId:
          type: string
          format: uuid
          description: ID счета
          example: "123e4567-e89b-12d3-a456-426614174000"
        type:
          type: string
          description: Тип транзакции
          enum: [DEPOSIT, WITHDRAWAL, TRANSFER, CREDIT_PAYMENT]
          example: "TRANSFER"
        amount:
          type: number
          format: decimal
          description: Сумма транзакции
          example: 500.00
        description:
          type: string
          description: Описание транзакции
          example: "Transfer to account 123456..."
        timestamp:
          type: string
          format: date-time
          description: Время транзакции
          example: "2024-01-15T14:30:00Z"
        balanceAfter:
          type: number
          format: decimal
          description: Баланс после транзакции
          example: 1000.00

    PageInfo:
      type: object
      description: Информация о пагинации
      properties:
        page:
          type: integer
          description: Текущая страница
          example: 0
        size:
          type: integer
          description: Размер страницы
          example: 20
        totalElements:
          type: integer
          description: Общее количество элементов
          example: 150
        totalPages:
          type: integer
          description: Общее количество страниц
          example: 8

    PagedResponseOfAccountResponse:
      type: object
      description: Страница со списком счетов
      properties:
        content:
          type: array
          description: Список счетов на текущей странице
          items:
            $ref: '#/components/schemas/AccountResponse'
        page:
          $ref: '#/components/schemas/PageInfo'

    PagedResponseOfTransactionResponse:
      type: object
      description: Страница со списком транзакций
      properties:
        content:
          type: array
          description: Список транзакций на текущей странице
          items:
            $ref: '#/components/schemas/TransactionResponse'
        page:
          $ref: '#/components/schemas/PageInfo'

    ErrorResponse:
      type: object
      description: Ответ с ошибкой
      properties:
        message:
          type: string
          description: Сообщение об ошибке
          example: "Insufficient funds"
        code:
          type: string
          description: Код ошибки
          example: "INSUFFICIENT_FUNDS"
        timestamp:
          type: string
          format: date-time
          description: Время ошибки
          example: "2024-01-15T14:30:00Z"
        path:
          type: string
          description: Путь запроса
          example: "/api/accounts/123/withdraw"

  securitySchemes:
    bearerAuth:
      type: http
      scheme: bearer
      bearerFormat: JWT
      description: JWT токен для аутентификации

security:
